#version 150
in vec3 position;
in vec2 texCoord;
in vec3 normal;

uniform mat4 mWorld;
uniform mat4 mView;
uniform mat4 mProjection;

out vec3 fNormal;
out vec3 fPosition;
out vec2 UV;
void main()
{
	//mat4 mModelView = mView * mWorld;
	vec4 pos = mWorld * vec4(position, 1.0);
	mat3 mWorldInvTrans=mat3(mWorld);
	fNormal = mWorldInvTrans * normal;
	fPosition = pos.xyz;
	gl_Position = mProjection * mView * pos;
	UV = texCoord;
}
